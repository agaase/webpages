var Genwi = Genwi || {};

/**
 * This is the helpers class object containing all the handlebars helper.
 * To register a specific set of helpers use the following statement.
 *
 * Genwi.Helpers.initialise(["helper1_name","helper2_name",..,])
 *
 * or do it individually: Genwi.Helpers.subset();
 * @namespace Genwi
 * @static
 * @class Helpers
 * @type {Object}
 */
Genwi.Helpers = {
    /**
     * Helper function to return the lowercase value for a string.
     *
     * @method lowercase
     * @return {[string]} the lowercase value of the value.
     */
    lowercase: function() {
        Handlebars.registerHelper('lowercase', function(value) {
            return value.toLowerCase();
        });
    },

    /**
     * Helper function to return the lowercase value for a variable.
     * @method uppercase
     * @return {[string]}  the uppercase string.
     */
    uppercase: function() {
        Handlebars.registerHelper('uppercase', function(value) {
            return value.toUpperCase();
        });
    },

    /**
     * This is a simple helper to get the substring for a string.
     * Works similar to javascript string function.
     * @method substring
     * @return {[string]}  [The result of substring operation.]
     */
    substring: function() {
        Handlebars.registerHelper("substring", function(string, start, end) {
            return (string.substring(start, end));
        });
    },

    /**
     * Return the different fields of time that can be arranged to get date in any format.
     * Use following fields to map different parts of the date.
     * date - The present date in number. e.g 29
     * day - Day in words e.g monday
     * month_num - The month in number. e.g 11
     * month -  Month in words e.g January
     * year - Year in number. e.g 2011
     * hour - Hours in 12 hr format.
     * min - Minutes.
     * sec - Seconds.
     * clock - AM/PM.
     * sup, - st/nd/th/rd to represent the date. e.g 1st.
     * @method datefields
     */
    datefields: function() {
        Handlebars.registerHelper("datefields", function(datestring, context) {
            var m_names = new Array("January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December");
            var d_names = new Array("Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday");
            var date;
            try {
                date = new Date(datestring);
            } catch (err) {
                console.log("Invalid format for string -'" + datestring + "'.Please check");
                return "";
            }

            function pad(arg) {
                arg += "";
                if (arg.length < 2) {
                    arg = "0" + arg;
                }
                return arg;
            }

            var fields = {};
            var date_num = date.getDate();
            fields.date = pad(date_num);
            fields.day = d_names[date.getDay()];
            var month_num = date.getMonth();
            fields.month_num = pad(month_num);
            fields.month = m_names[month_num];
            fields.year = date.getFullYear();
            fields.hour = pad(date.getHours());
            fields.min = pad(date.getMinutes());
            fields.sec = pad(date.getSeconds());

            fields.clock = "AM";
            if (fields.hour > 11) {
                fields.clock = "PM";
                if (fields.hour > 12) {
                    fields.hour = fields.hour - 12;
                }
            }
            if (date_num === 1 || date_num === 21 || date_num === 31) {
                fields.sup = "st";
            } else if (date_num === 2 || date_num === 22) {
                fields.sup = "nd";
            } else if (date_num === 3 || date_num === 23) {
                fields.sup = "rd";
            } else {
                fields.sup = "th";
            }
            var buffer = context.fn(fields) || "";
            return buffer;
        });
    },

    /**
     * This is a simple helper to split a string using a seperator
     * and then apply the array to the template.
     * @method split
     * @return {htmlstring} The compiled html.
     */
    split: function() {
        Handlebars.registerHelper("split", function(string, seperator, context) {
            var buffer = "";

            var items = string.split(seperator);
            var data = {};
            data.words = items;

            buffer = context.fn(data);
            return buffer;
        });
    },

    /**
     * Strips a text from its html formatting. Uses the jquery function to do so.
     * We also wrap the text here to allow some simple string be in the argument.
     * @method rawtext
     * @return {[htmltext]}  The raw string without any html tags.
     */
    rawtext: function() {
        Handlebars.registerHelper("rawtext", function(value) {
            return ($('<i>').html(value).text());
        });
    },

    /**
     * This is a helper function to iterate over a an array for a particular subset.
     * @param  {[object]} The array object
     * @param  {[number]} The start position
     * @param  {[number]} the number by which the length should be shortened to
     * @param  {[object]} The context which has the template to apply
     * @return {[htmltext]}
     */
    subset: function() {
        Handlebars.registerHelper("subset", function(array, start, shortcount, context) {
            var buffer = "";

            if (array && array.length > 0) {
                var end = array.length - shortcount - 1;

                for (var i = start; i <= end; i++) {
                    var item = array[i];
                    buffer += context.fn(item);
                }
            }

            return buffer;
        });
    },


    /**
     * This helper compares two value depending on the operator type passed.
     * If the result is true applies the template inside compare otherwise applies the else part.
     * @method compare
     * @param {object} The left value to compare.
     * @param {object} The right hand side value to compare.
     * @param {string} type The operator type - eq or neq.
     */
    compare: function() {
        Handlebars.registerHelper("compare", function(argLeft, argRight, type, context) {
            var buffer = "";
            if (type === "eq" && argLeft === argRight) {
                buffer = context.fn(this);
            } else if (type === "neq" && argLeft !== argRight) {
                buffer = context.fn(this);
            } else {
                buffer = context.inverse(this);
            }
            return buffer;
        });
    },

    /**
     * Initialises the helper using the list of helpers passed.
     * Pass an empty array or no argument if you want all the helpers to be registered.
     * @method initialise
     * @param  {[array]}   helpers- Comma seperated list of the name of the helpers.
     */
    initialise: function(helpers) {

        var helpObj = this;

        if (helpers && helpers.length > 0) {
            $.each((helpers), function(index, value) {
                if (!(value in Handlebars.helpers)) {
                    helpObj[value]();
                }
            });
        } else {
            $.each(helpObj, function(name, value) {
                if (name !== "initialise" && !(name in Handlebars.helpers)) {
                    helpObj[name]();
                }
            });
        }
    }


};
