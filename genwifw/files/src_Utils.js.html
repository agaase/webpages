    var Genwi = Genwi || {};

/**
 * This class contains plain old school utility functions.
 *
 * Usage is pretty simple.E.g. Genwi.Utils.getDateStr();
 * @namespace Genwi
 * @class Utils
 * @type {Object}
 */
Genwi.Utils = {

    /**
     * Utility to get date in particular format.
     * @method getDateStr
     * @param  {[date]} date
     * @return {[string]} date in form "January 02 2012"
     */
    getDateStr: function(date) {
        var monthNames = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"];
        var day = date.getDate();
        var month = date.getMonth();
        var year = date.getFullYear();
        return monthNames[month] + " " + day + " " + year;
    },
    /**
     * This utility is to load the images after the layout is done.
     * Image tags or div tag(background image) should fullfill following requirement
     *
     *   1. data-src attribute set to url of the image
     *
     *   e.g Genwi.Utils.lazyLoad($(".lazyElements"));
     * @method lazyLoad
     * @param  {array} lazyElements
     *
     */
    lazyLoad: function(lazyElements) {
        function setImg(type, ele, src) {
            if (type === "div") {
                ele.css('background-image', 'url(' + src + ')');
                var loader = $(".loader", ele);
                if (loader[0]) {
                    $(loader[0]).remove();
                }
            } else {
                ele.attr("src", src);
            }
            ele.removeClass("oLazy");
            ele.addClass("lazyLoaded");
        }
        lazyElements.each(function(index) {
            var ele = $(this);
            var src = "" + ele.attr("data-src");
            if (src && src.length > 0) {
                var type = ele.attr("tagName");
                if (type === undefined) {
                    //For new version of jquery.
                    type = ele.prop("tagName");
                }
                type = type.toLowerCase();
                $("<img/>")
                    .load(function() {
                        setImg(type, ele, src);
                    })
                    .ready(function() {
                        setImg(type, ele, src);
                    })
                    .error(function() {
                        console.log("Error loading image");
                    })
                    .attr("src", src);
            }
        });
    },

    /**
     * Fills up a basic handlebar template.
     * @method bindTmplData
     * @param  {[string]} templateId -  The Id of the handlebar template
     * @param  {[string(json)]} context - The context to fill up the template.
     * @param  {[string(jquery-selector)]} elementToAppend - The element where the final layout should be appended to.
     * @param {boolean} [append] [Append original contents. True to append, false otherwise.]
     *
     */
    bindTmplData: function(templateSc, model, elementToAppend, append) {
        if (templateSc.length) {
            var template = Handlebars.compile(templateSc.html());
            var layout = template(model);
            if (append) {
                $("'" + layout + "'").appendTo(elementToAppend);
            } else {
                elementToAppend.html(layout);
            }
        }
    }

};
